\section{Linear Search}
Algorithm to find index of the element
\begin{algorithm}
    \caption{Linear Search}
\KwIn{An array $A$ of $n$ size and element $x$}
\KwOut{Returns index of $x$ , otherwise $-1$}
\For{$i \gets 1$ \KwTo n }{
  \If{$A[i] = x$}{
    \Return $i$\
  }
}
 \Return $-1$\
\end{algorithm}
